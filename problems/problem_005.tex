
\begin{exercise}[The orthogonal complement of the orthogonal complement]
    Let $V$ be a finite-dimensional vector space. Let $W$ be a subspace of $V$. 
    
    Show that the orthogonal complement of the orthogonal complement of $W$, denoted by $(W^{\perp})^{\perp}$, is equal to $W$
    
\end{exercise}
