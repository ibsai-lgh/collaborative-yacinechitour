
\begin{exercise}[Lemma: finding a vector outside of two subspaces]
    Let $V$ be a finite-dimensional vector space. Let $U$ and $W$ be subspaces of $V$ such that $\dim(U) < \dim(V)$ and $\dim(W) < \dim(V)$.
    
    Show that there exists a vector $v \in V$ such that $v \notin U$ and $v \notin W$.

\end{exercise}
